 86     parse:function(q) {
 87       var queryString = q.replace(/(\s+)?(=|<=|>=|<>|<|>| LIKE | like )(\s+)?/g,"$2").replace(/""|''/g,"Null").replace(/==/g,"="); // remove unnecessary white space & replace ' 
 88       var factory = [];
 89       var limitMax = q.length;
 90       var closeOperator="", closeTag = "";
 91       var lastField = "";
 92       var parenthesis = {open:0};
 93       for (var i=0; i < queryString.length; i++) {
 94         var letter = queryString.charAt(i);
 95         switch (letter) {
 96           case "(": // find the deepest (
 97                     var start = i;
 98                     var openedApos=false;
 99                     while (queryString.charAt(i) == "(" && i < limitMax) { i++; parenthesis.open++; }
100                     // find the corresponding )
101                     while (parenthesis.open>0 && i < limitMax) {
102                       i++;
103                       // if there is a ' opened then ignore the ) until the next '
104                       var charAtI = queryString.charAt(i);
105                       if (charAtI=="'") openedApos=!openedApos;
106                       else if (charAtI==")" && !openedApos) parenthesis.open--;
107                     }
108                     
109                     var lastIndex = factory.length-1;
110   
111                     // concat with the first index
112                     if (lastIndex>=0) {
113                       if (closeOperator != "") factory[0] = "<"+closeOperator+">"+factory[0];
114                       factory[0] += this.parse(queryString.substring(start+1, i));
115                       if (closeOperator != "") factory[0] += "</"+closeOperator+">";
116                       //delete(factory[lastIndex]);
117                       closeOperator = "";
118                     } else factory[0] = this.parse(queryString.substring(start+1, i));
119                     break;
120           case ">":  // look at the operand
121           case "<": i++;
122                     if (queryString.charAt(i) == "=") { // >= or <=
123                       factory.push("<"+(letter==">"?"G":"L")+"eq>");
124                       closeTag = "</"+(letter==">"?"G":"L")+"eq>";
125                     } else if (letter == "<" && queryString.charAt(i) == ">") { // <>
126                       factory.push("<Neq>");

128                     } else {

132                     }
133                     break;
134           case "=": factory.push("<Eq>");
135                     closeTag = "</Eq>";
136                     break;
137           case " ": // check if it's AND or OR
